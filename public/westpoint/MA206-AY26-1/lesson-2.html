What is synthetic data?

Code 
library(tidyverse)
set.seed(206)
n <- 300

df <- 
  tibble(id = 1:n) |> 
  mutate(# Categorical variables
    sport  = as.factor(sample(c("Handball", "Lacrosse", "Track"), size = n, replace = TRUE, prob = c(0.33, 0.34, 0.33))),
    gender = as.factor(sample(c("Female", "Male"), size = n, replace = TRUE, prob = c(0.45, 0.55)))
  ) |> 
  mutate(# Quantitative variables
    height_cm = round(rnorm(n, mean = ifelse(gender == "Male", 178, 166) + ifelse(sport == "Track", -1.0, ifelse(sport == "Lacrosse", 0.5, 0)), sd = 6.5), 1),
    training_hours = pmax(0, round(rnorm(n, mean = ifelse(sport == "Track", 7.5, ifelse(sport == "Lacrosse", 6.0, 5.5)), sd = 2.0), 1)
    )
  ) |> 
  mutate(# Response variable (run time in minutes)
    run_time_min = 16.5 + 
      ifelse(sport == "Handball", 1.0, 
             ifelse(sport == "Lacrosse", 0.6, -1.4)) + 
      ifelse(gender == "Female", 0.7, 0) + (-0.22) * training_hours + (0.015) * height_cm + 
      ifelse(sport == "Track", -0.05 * training_hours, 0) + rnorm(n, 0, 0.8), run_time_min = round(pmin(pmax(run_time_min, 11), 24), 2)
       )
